3   import java.util.List;
4   import java.util.Map;
5   
6   /**
7    * general class for making pairs.
8    */
9   public abstract class PairGenerator {
10    /**
11     * This maps a person to a set of their preferences, sorted from highest preference to
12     * lowest preference.
13     * <p>
14     * The value should refer to the same list in memory as the key's preferences list.
15     */
16    protected final Map<Person, List<Person>> personToPreferences;
17  
18    /**
19     * @param personToPreferences - a map of persons to their preferences
20     */
21    public PairGenerator(Map<Person, List<Person>> personToPreferences) {
22      this.personToPreferences = personToPreferences;
23    }
24  
25    /**
26     * @return - a map that represents pairs
27     */
28    public abstract Map<Person, Person> getPairs();
29  
30    /**
31     * @param map - a map that represents pairs
32     * @return - true if the pairs are stable, false otherwise
33     */
34    public boolean isStable(Map<Person, Person> map) {
35      if (map == null) {
36        return false;
37      }
38  
39      // if someone doesn't have a partner
40      for (Person currPerson : map.keySet()) {
41        if (map.get(currPerson) == null) {
42          return false;
43        }
44      }
45  
46      // someone's partner's partner should be themselves and their partner should not be themselves
47      for (Person currPerson : map.keySet()) {
48        if (!map.get(map.get(currPerson)).equals(currPerson)
49            || map.get(currPerson).equals(currPerson)) {
50          return false;
51        }
52      }
53  
54      return true;
55    }
56  }
