3   import java.util.List;
4   import java.util.Map;

6   public abstract class PairGenerator {
7     /**
8      * This maps a person to a set of their preferences, sorted from highest preference to
9      * lowest preference.
10     * <p>
11     * The value should refer to the same list in memory as the key's preferences list.
12     */
13    @SuppressWarnings("checkstyle:VisibilityModifier")
14    protected final Map<Person, List<Person>> personToPreferences;
15  
16    public PairGenerator(Map<Person, List<Person>> personToPreferences) {
17      this.personToPreferences = personToPreferences;
18    }
19  
20    public abstract Map<Person, Person> getPairs();
21  
22    public boolean isStable(Map<Person, Person> map) {
23      if (map == null) {
24        return false;
25      }
26  
27      // if someone doesn't have a partner
28      for (Person currPerson : map.keySet()) {
29        if (map.get(currPerson) == null) {
30          return false;
31        }
32      }
33  
34      // someone's partner's partner should be themselves and their partner should not be themselves
35      for (Person currPerson : map.keySet()) {
36        if (!map.get(map.get(currPerson)).equals(currPerson) || map.get(currPerson).equals(currPerson)) {
37          return false;
38        }
39      }
40  
41      return true;
42    }
43  }
