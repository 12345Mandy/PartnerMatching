15  /**
16   * class for converting answer choices into preferences.
17   */
18  public class MatchEvaluator {
19    private List<Question> questions; // a list of the questions
20    private Map<Person, List<Answer>> personToAnswers; // a map from person to their answers
21  
22    /**
23     * @param questions       - the list of the questions that were asked
24     * @param personToAnswers - a map from a person to their answer choices
25     */
26    public MatchEvaluator(List<Question> questions, Map<Person, List<Answer>> personToAnswers) {
27      this.questions = questions;
28      this.personToAnswers = personToAnswers;
29  
30    }
31  
32    /**
33     * Gets all the info -- answers from each user who took survey -- evaluates similarity.
34     *
35     * @return a hashmap mapping ids to a list of their matches sorted
36     */
37    public Map<Person, List<Person>> evaluateMatches() {
38      Map<Person, List<Person>> finalMatches = new HashMap<>();
39      List<Question> questionInfo = this.questions;
40  
41      // starts through every person
42      for (Person currPerson : this.personToAnswers.keySet()) {
43        // stores list of ids that represent the similarity
44        List<Pair<Person, Double>> idScorePairs = new ArrayList<>();
45        List<Answer> currPersonsAnswers = this.personToAnswers.get(currPerson);
46  
47        // goes through each person who took the survey
48        // if not the same person, calculate score
49        // score will be [id, score] and put into similarity array
50        for (Person potMatch : this.personToAnswers.keySet()) {
51          if (!currPerson.equals(potMatch)) {
52            List<Answer> pot = this.personToAnswers.get(potMatch);
53            double potScore = getScore(currPersonsAnswers, pot, questionInfo);
54  
55            Pair<Person, Double> pair = new Pair<Person, Double>(potMatch, potScore);
56            idScorePairs.add(pair);
57          }
58        }
59  
60        // sort the array of [id, score] pairs based off score
61        // when sorted --> go through each element and add to list
62        // so this will give us a list of IDs based on order of score
63        idScorePairs.sort((a, b) -> Double.compare(b.getSecond(), a.getSecond()));
64        List<Person> sortedIDs = new ArrayList<>();
65        for (Pair<Person, Double> pairs : idScorePairs) {
66          sortedIDs.add(pairs.getFirst());
67        }
68  
69        // add to hashmap
70        // will have ID --> List of sorted IDs
71        // the first id being most similar
72        finalMatches.put(currPerson, sortedIDs);
73      }
74      return finalMatches;
75    }
76  
77    /**
78     * Calculates similarity score based off two people and the points for each question.
79     *
80     * @param person1answers - list of answers one user had
81     * @param person2answers - list of answers another user had
82     * @param questionList   - a list of questions; score of each question -- add to total score
83     *                       if answers are the same
84     * @return total score
85     */
86    public double getScore(List<Answer> person1answers, List<Answer> person2answers,
87                           List<Question> questionList) {
88      int totalScore = 0;
89      for (int i = 0; i < person1answers.size(); i++) {
90        if (person1answers.get(i).equals(person2answers.get(i))) {
91          totalScore += questionList.get(i).getScore();
92        }
93      }
94  
95      return totalScore;
96    }
97  }
