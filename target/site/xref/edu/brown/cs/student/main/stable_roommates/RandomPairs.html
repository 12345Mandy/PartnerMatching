4   import java.util.HashMap;
5   import java.util.List;
6   import java.util.Map;

9   public class RandomPairs extends PairGenerator {
10    public RandomPairs(Map<Person, List<Person>> personToPreferences) {
11      super(personToPreferences);
12    }
13  
14    public Map<Person, Person> getPairs() {
15      List<Person> people = new ArrayList<>(personToPreferences.keySet());
16  
17      assert people.size() % 2 == 0;
18      Map<Person, Person> toReturn = new HashMap<>();
19      while (people.size() != 0) {
20        Random rand = new Random();
21  
22        int indexOne = rand.nextInt(people.size());
23        int indexTwo = rand.nextInt(people.size());
24  
25        if (indexOne == indexTwo) {
26          continue;
27        }
28  
29        Person first = people.get(indexOne);
30        Person second = people.get(indexTwo);
31  
32        people.remove(first);
33        people.remove(second);
34  
35        toReturn.put(first, second);
36        toReturn.put(second, first);
37      }
38  
39      return toReturn;
40    }
41  }
